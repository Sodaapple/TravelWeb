	11. 複習資料庫指令
		
		-- 310 列出 以下欄位 ProductID, ProductName, SupplierID
		

		-- 315 延續上題 310
		--     請帶出該產品的供應商公司名稱(CompanyName)、聯絡電話(Phone)、聯絡人(ContactName)
		--     相同供應商的資料請列在一起
		select A.ProductID, A.ProductName,A.SupplierID,A.UnitsInStock,B.CompanyName,B.Phone,B.ContactName from  Products as A join Suppliers as B on A.SupplierID=B.SupplierID

		-- 320 延續上題 310
		--     請加入條件："庫存量低於再訂購量" 的產品資料

		select A.ProductID, A.ProductName,A.SupplierID,A.UnitsInStock,A.UnitsOnOrder,B.CompanyName,B.Phone,B.ContactName from  Products as A join Suppliers as B on A.SupplierID=B.SupplierID 
		where UnitsInStock<UnitsOnOrder

		select * from  Products  where UnitsInStock=0
		select * from  Suppliers
		
		-- 340. 那張訂單賺最多錢？ 

		select TOP 1 OrderID,sum(UnitPrice* Quantity) as A from [Order Details] group by  OrderID order by A desc 


		-- 350. 總共有幾位業務負責處理VINET的訂單？

		-- 360. 負責處理VINET的訂單是哪些業務？
